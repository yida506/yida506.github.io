重载的代码hook

以该处为例，该类中有两个a方法，我们只想hook上面一个，直接写就会报错。

这样，我们只需要等他报错后，修改代码即可。

// 加上.overload('a.u$a')
Signature.a.overload('a.u$a').implementation = function(val){
            console.log('enter a');    
            console.log(val);
            return this.a(val);
        };

以上是对于调用的函数进行hook，得到的，还有一种情况，就是方法没有被调用，我们想查看他的方法，拿到其结果，那么我们可以采用Java.choose(类名+回调)。

function hook() {
    Java.perform(function () {
        const commonPaths = [
            "/data/local/bin/su",
            "/data/local/su",

            "/dev/com.koushikdutta.superuser.daemon/",
            "/sbin/su",

        ];

        var JavaString = Java.use("java.lang.String");
        JavaString.contains.implementation = function (name) {
            if (name !== "test-keys") {
                var ret = this.contains(name);
                console.log("JavaString",name,ret);
            }
            else{
                var ret = this.contains(name);
                return false
            }
            return ret
        };

        var JavaRuntime = Java.use("java.lang.Runtime");
        var iOException = Java.use("java.io.IOException");
        JavaRuntime.exec.overload("java.lang.String").implementation = function (command) {
            if (command.endsWith("su")) {
                ret = this.exec(command);
                console.log("JavaRuntime" ,command,ret);
                throw iOException.$new("Hacker");
            }
            else {
                var ret = this.exec(command);
                console.log("JavaRuntime2" ,command,ret)
            }
            return ret;
        };

        var JavaFile = Java.use("java.io.File");
        JavaFile.exists.implementation = function () {
            const filename = this.getAbsolutePath();
            if (commonPaths.indexOf(filename) >= 0) {
                var ret = this.exists();
                console.log("JavaFile",filename,ret);
                return false
            }
            else {
                var ret = this.exists();
            }
            return ret;

};

setImmediate(function () {

Objection

Objection本质上是frida的一种工具实现，可以通过命令行输出，实现frida交互过程，达到类调用，hook的目的。

安装

由于frida版本是14.1.3，直接pip install objection即可

使用
// 注入代码
objection -g 进程名 explore  --startup-command "android root disable" <在启动前注入命令>
  1. 寻找包名 frida-ps -U | grep 参数
  2. Objection注入
  3. android hooking search classes[类名]
  4. android hooking search methods [关键词] (从所有内存中加载)
  5. android hooking list class methods 包名+类名 (列出所有的方法)
  6. android hooking list 组件
  7. android hooking watch class_method (hook指定方法) –dump-args –dump-backtrace –dump-return
  8. jobs list (查看作业，显示hook的函数的信息)
  9. job kill 删除作业

如果要hook构造方法 需要使用$init
